- new viz will call a method in visualizations to check if all methods in an interface are implemented
- call visualizations.method(vizObj) will call vizObj.method()
    - packed all methods into one `functions` object

// Check if the object implements a function with the provided signature
var isImplemented = function(object, name, length) {
return (typeof object[name] === 'function') &&
    (object[name].length === length);
}
